\subsection{Practice 5}

\noindent \hspace{1.2em}\textit{Find the intervals where the following functions are convex upward or convex downward, and find the points of inflection of the functions:}
\begin{enumerate}
    \begin{multicols}{2}
        \item $f(x)=3 x^2-x^3$
        \sol{}
        \begin{flalign*}
            f'(x)  & = 6x - 3x^2          & \\
            f''(x) & = 6 - 6x             & \\
            6 - 6x & = 0                  & \\
            x      & = 1                  & \\
            f(1)   & = 3(1)^2 - (1)^3 = 2
        \end{flalign*}
        In the interval $(-\infty, 1)$, $f''(x) > 0$, so the function is convex downward.

        In the interval $(1, \infty)$, $f''(x) < 0$, so the function is convex upward.

        The point of inflection is $(1, 2)$.

        \item $f(x)=\dfrac{1}{3} x^3-x^2-3 x+2$
        \sol{}
        \begin{flalign*}
            f'(x)  & = x^2 - 2x - 3                                         & \\
            f''(x) & = 2x - 2                                               & \\
            2x - 2 & = 0                                                    & \\
            x      & = 1                                                    & \\
            f(1)   & = \dfrac{1}{3}(1)^3 - (1)^2 - 3(1) + 2 = -\dfrac{5}{3}
        \end{flalign*}
        In the interval $(-\infty, 1)$, $f''(x) < 0$, so the function is convex upward.

        In the interval $(1, \infty)$, $f''(x) > 0$, so the function is convex
        downward.
    \end{multicols}
\end{enumerate}
