\section*{Exercise 11j}

Find the following indefinite integrals:
\begin{enumerate}
      \item $\displaystyle\int x^2e^{-x}\,dx$
            \sol{}

            Let $u = x^2$, $du = 2x\,dx$.

            Let $dv = e^{-x}\,dx$, $v = -e^{-x}$.
            \begin{flalign*}
                  \int x^2e^{-x}\,dx & = -x^2e^{-x} + \int 2xe^{-x}\,dx                        & \\
                                     & = -x^2e^{-x} + 2\int xe^{-x}\,dx                        & \\
                                     & = -x^2e^{-x} + 2\left(-xe^{-x} + \int e^{-x}\,dx\right) & \\
                                     & = -x^2e^{-x} + 2\left(-xe^{-x} - e^{-x}\right) +C       & \\
                                     & = -x^2e^{-x} - 2xe^{-x} - 2e^{-x} + C                   & \\
                                     & = -(x^2 + 2x + 2)e^{-x} + C \eos
            \end{flalign*}

      \item $\displaystyle\int x^2\cos x\,dx$
            \sol{}

            Let $u = x^2$, $du = 2x\,dx$.

            Let $dv = \cos x\,dx$, $v = \sin x$.
            \begin{flalign*}
                  \int x^2\cos x\,dx & = x^2\sin x - \int 2x\sin x\,dx                        & \\
                                     & = x^2\sin x + 2\int x\sin x\,dx                        & \\
                                     & = x^2\sin x + 2\left(-x\cos x - \int \cos x\,dx\right) & \\
                                     & = x^2\sin x + 2\left(-x\cos x - \sin x\right) +C       & \\
                                     & = x^2\sin x - 2x\cos x - 2\sin x + C                   & \\
                                     & = (x^2 - 2)\sin x - 2x\cos x + C \eos
            \end{flalign*}

      \item $\displaystyle\int x^2\cos^2 x\,dx$
            \sol{}

            Let $u = x^2$, $du = 2x\,dx$. \vspace{-0.5em}
            \begin{flalign*}
                  \text{Let }dv = \cos^2 x\,dx,\ v & = \int \cos^2 x\,dx                 & \\
                                                   & = \frac{1}{2}\int (1 + \cos 2x)\,dx & \\
                                                   & = \frac{1}{2}x + \frac{1}{4}\sin 2x
            \end{flalign*}
            \begin{flalign*}
                  \int x^2\cos^2 x\,dx & = x^2\left(\frac{1}{2}x + \frac{1}{4}\sin 2x\right) - \int 2x\left(\frac{1}{2}x + \frac{1}{4}\sin 2x\right)\,dx                               & \\
                                       & = \dfrac{1}{2}x^3 + \dfrac{1}{4}x^2\sin 2x - \int \left(x^2 + \dfrac{1}{2}x\sin 2x\right)\,dx                                                 & \\
                                       & = \dfrac{1}{2}x^3 + \dfrac{1}{4}x^2\sin 2x - \int x^2\,dx - \dfrac{1}{2}\int x\sin 2x\,dx                                                     & \\
                                       & = \dfrac{1}{2}x^3 + \dfrac{1}{4}x^2\sin 2x - \dfrac{1}{3}x^3 - \dfrac{1}{2}\left(-\dfrac{1}{2}x\cos 2x - \int -\dfrac{1}{2}\cos 2x\,dx\right) & \\
                                       & = \dfrac{1}{6}x^3 + \dfrac{1}{4}x^2\sin 2x + \dfrac{1}{4}x\cos 2x - \dfrac{1}{4}\int \cos 2x\,dx                                              & \\
                                       & = \dfrac{1}{6}x^3 + \dfrac{1}{4}x^2\sin 2x + \dfrac{1}{4}x\cos 2x - \dfrac{1}{8}\sin 2x + C \eos
            \end{flalign*}

      \item $\displaystyle\int x^5\sin x^2\,dx$
            \sol{}

            Let $u = x^4$, $du = 4x^3\,dx$.

            Let $dv = x\sin x^2\,dx$, $v = -\dfrac{1}{2}\cos x^2$.
            \begin{flalign*}
                  \int x^5\sin x^2\,dx & = -\dfrac{1}{2}x^4\cos x^2 + \int 2x^3\cos x^2\,dx & \\
                                       & = -\dfrac{1}{2}x^4\cos x^2 + 2\int x^3\cos x^2\,dx
            \end{flalign*}
            Let $u = x^2$, $du = 2x\,dx$.

            Let $dv = x\cos x^2\,dx$, $v = \dfrac{1}{2}\sin x^2$.
            \begin{flalign*}
                  \int x^5\sin x^2\,dx & = -\dfrac{1}{2}x^4\cos x^2 + 2\left(\dfrac{1}{2}x^2\sin x^2 - \int \dfrac{1}{2} \cdot 2x\sin x^2\,dx\right) & \\
                                       & = -\dfrac{1}{2}x^4\cos x^2 + x^2\sin x^2 - 2\int x\sin x^2\,dx                                              & \\
                                       & = -\dfrac{1}{2}x^4\cos x^2 + x^2\sin x^2 + \cos x^2 + C \eos
            \end{flalign*}

      \item $\displaystyle\int\sin(\ln x)\,dx$
            \sol{}

            Let $t = \ln x$, $x = e^t$, $dx = e^t\,dt$.
            \begin{flalign*}
                  \int\sin(\ln x)\,dx & = \int\sin t\cdot e^t\,dt & \\
                                      & = \int e^t\sin t\,dt      & \\
            \end{flalign*}
            Let $u = e^t$, $du = e^t\,dt$.

            Let $dv = \sin t\,dt$, $v = -\cos t$.
            \begin{flalign*}
                  \int\sin(\ln x)\,dx & = -e^t\cos t - \int -\cos t \cdot e^t\,dt & \\
                                      & = -e^t\cos t + \int e^t\cos t\,dt         &
            \end{flalign*}
            Let $u = e^t$, $du = e^t\,dt$.

            Let $dv = \cos t\,dt$, $v = \sin t$.
            \begin{flalign*}
                  \int\sin(\ln x)\,dx & = -e^t\cos t + e^t\sin t - \int e^t\sin t\,dt                  & \\
                                      & = \dfrac{1}{2}e^t(\sin t - \cos t) +C                          & \\
                                      & = \dfrac{1}{2}x\left[\sin(\ln x) - \cos(\ln x)\right] + C \eos
            \end{flalign*}

      \item $\displaystyle\int e^x\cos x\,dx$
            \sol{}

            Let $u = e^x$, $du = e^x\,dx$.

            Let $dv = \cos x\,dx$, $v = \sin x$.
            \begin{flalign*}
                  \int e^x\cos x\,dx & = e^x\sin x - \int e^x\sin x\,dx &
            \end{flalign*}
            Let $u = e^x$, $du = e^x\,dx$.

            Let $dv = \sin x\,dx$, $v = -\cos x$.
            \begin{flalign*}
                  \int e^x\cos x\,dx & = e^x\sin x + e^x\cos x - \int e^x\cos x\,dx & \\
                                     & = \dfrac{1}{2}e^x(\sin x + \cos x) + C \eos
            \end{flalign*}

      \item $\displaystyle\int x^5e^{x^2}\,dx$
            \sol{}

            Let $u = x^4$, $du = 4x^3\,dx$.

            Let $dv = xe^{x^2}\,dx$, $v = \dfrac{1}{2}e^{x^2}$.
            \begin{flalign*}
                  \int x^5e^{x^2}\,dx & = \dfrac{1}{2}x^4e^{x^2} - \int 2x^3e^{x^2}\,dx & \\
                                      & = \dfrac{1}{2}x^4e^{x^2} - 2\int x^3e^{x^2}\,dx
            \end{flalign*}
            Let $u = x^2$, $du = 2x\,dx$.

            Let $dv = xe^{x^2}\,dx$, $v = \dfrac{1}{2}e^{x^2}$.
            \begin{flalign*}
                  \int x^5e^{x^2}\,dx & = \dfrac{1}{2}x^4e^{x^2} - 2\left(\dfrac{1}{2}x^2e^{x^2} - \int \dfrac{1}{2} \cdot 2xe^{x^2}\,dx\right) & \\
                                      & = \dfrac{1}{2}x^4e^{x^2} - x^2e^{x^2} + 2\int xe^{x^2}\,dx                                              & \\
                                      & = \dfrac{1}{2}x^4e^{x^2} - x^2e^{x^2} + e^{x^2} + C                                                     & \\
                                      & = \dfrac{1}{2}e^{x^2}(x^4 - 2x^2 + 2) + C \eos
            \end{flalign*}

      \item $\displaystyle\int e^{2x}\cos x\,dx$
            \sol{}

            Let $u = e^{2x}$, $du = 2e^{2x}\,dx$.

            Let $dv = \cos x\,dx$, $v = \sin x$.
            \begin{flalign*}
                  \int e^{2x}\cos x\,dx & = e^{2x}\sin x - 2\int e^{2x}\sin x\,dx &
            \end{flalign*}
            Let $u = e^{2x}$, $du = 2e^{2x}\,dx$.

            Let $dv = \sin x\,dx$, $v = -\cos x$.
            \begin{flalign*}
                  \int e^{2x}\cos x\,dx & = e^{2x}\sin x - 2\left(-e^{2x}\cos x - 2\int -e^{2x}\cos x\,dx\right) & \\
                                        & = e^{2x}\sin x + 2e^{2x}\cos x - 4\int e^{2x}\cos x\,dx                & \\
                                        & = \dfrac{1}{5}e^{2x}(\sin x + 2\cos x) + C \eos
            \end{flalign*}

      \item $\displaystyle\int (x^2 + 7x - 5)\cos2x\,dx$
            \sol{}

            Let $u = x^2 + 7x - 5$, $du = (2x + 7)\,dx$.

            Let $dv = \cos2x\,dx$, $v = \dfrac{1}{2}\sin2x$.
            \begin{flalign*}
                  \int (x^2 + 7x - 5)\cos2x\,dx & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x - \int \dfrac{1}{2}\sin2x(2x + 7)\,dx                         & \\
                                                & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x - \dfrac{1}{2}\int 2x\sin2x\,dx - \dfrac{7}{2}\int \sin2x\,dx & \\
                                                & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x - \int x\sin2x\,dx + \dfrac{7}{4}\cos2x                       &
            \end{flalign*}
            Let $u = x$, $du = dx$.

            Let $dv = \sin2x\,dx$, $v = -\dfrac{1}{2}\cos2x$.
            \begin{flalign*}
                  \int (x^2 + 7x - 5)\cos2x\,dx & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x - \left(-\dfrac{1}{2}x\cos2x + \dfrac{1}{2}\int\cos2x\,dx\right) + \dfrac{7}{4}\cos2x & \\
                                                & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x + \dfrac{1}{2}x\cos2x - \dfrac{1}{2}\int\cos2x\,dx + \dfrac{7}{4}\cos2x               & \\
                                                & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x + \dfrac{2}{4}x\cos2x - \dfrac{1}{4}\sin2x + \dfrac{7}{4}\cos2x + C                   & \\
                                                & = \dfrac{1}{2}(x^2 + 7x - 5)\sin2x + \dfrac{1}{4}(2x + 7)\cos2x - \dfrac{1}{4}\sin2x + C \eos
            \end{flalign*}

      \item $\displaystyle\int e^{ax}\sin bx\,dx$
            \sol{}

            Let $u = e^{ax}$, $du = ae^{ax}\,dx$.

            Let $dv = \sin bx\,dx$, $v = -\dfrac{1}{b}\cos bx$.
            \begin{flalign*}
                  \int e^{ax}\sin bx\,dx & = -\dfrac{1}{b}e^{ax}\cos bx + \dfrac{a}{b}\int e^{ax}\cos bx\,dx &
            \end{flalign*}
            Let $u = e^{ax}$, $du = ae^{ax}\,dx$.

            Let $dv = \cos bx\,dx$, $v = \dfrac{1}{b}\sin bx$.
            \begin{flalign*}
                  \int e^{ax}\sin bx\,dx & = -\dfrac{1}{b}e^{ax}\cos bx + \dfrac{a}{b}\left(\dfrac{1}{b}e^{ax}\sin bx - \dfrac{a}{b}\int e^{ax}\sin bx\,dx\right) & \\
                                         & = -\dfrac{1}{b}e^{ax}\cos bx + \dfrac{a}{b^2}e^{ax}\sin bx - \dfrac{a^2}{b^2}\int e^{ax}\sin bx\,dx                    & \\
                                         & = \dfrac{b^2}{b^2 + a^2}\left(\dfrac{a}{b^2}e^{ax}\sin bx - \dfrac{1}{b}e^{ax}\cos bx\right) + C                         \\
                                         & = \dfrac{b^2}{a^2 + b^2} \cdot \dfrac{e^{ax}}{b^2}\left(a\sin bx - b\cos bx\right) + C                                   \\
                                         & = \dfrac{e^{ax}}{b^2 + a^2}\left(a\sin bx - b\cos bx\right) + C \eos
            \end{flalign*}
\end{enumerate}
