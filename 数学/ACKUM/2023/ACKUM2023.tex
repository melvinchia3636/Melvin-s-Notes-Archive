    \item The inverse function of $f(x) = 1 - \sqrt{4 - 3x}$, $0 \leq x \leq 4$ is $g(x) = px^2 + qx + r$, $a \leq x \leq b$. Find the value of $r + b$.

          Let $u = f^{-1}(x)$, $f(u) = x$
          \begin{align*}
              1 - \sqrt{4 - 3u} & = x                           \\
              - \sqrt{4 - 3u}   & = x - 1                       \\
              4 - 3u            & = x^2 - 2x + 1                \\
              -3u               & = x^2 - 2x - 3                \\
              u                 & = -\dfrac{1}{3}(x^2 - 2x - 3)
          \end{align*}

\end{enumerate}
