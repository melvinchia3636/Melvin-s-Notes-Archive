\begin{enumerate}
    \item Find the gradient of the tangent to the curve $y = 2x^2 + 1$ at the point where
          $x = 2$. \sol{}
          \begin{flalign*}
              \Delta y                                        & = 2{(x + \Delta x)}^2 + 1 - (2x^2 + 1)             \\
                                                              & = 2x^2 + 4x\Delta x + 2{\Delta x}^2 + 1 - 2x^2 - 1 \\
                                                              & = 4x\Delta x + 2{\Delta x}^2                       \\
              \\
              \frac{\Delta y}{\Delta x}                       & = 4x + 2\Delta x                                   \\
              \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} (4x + 2\Delta x)    \\
              \frac{dy}{dx}                                   & = 4x
          \end{flalign*}
          $\therefore$ The gradient of the tangent to the curve $y = 2x^2 + 1$ at the point where $x = 2$ is $4(2) = 8$.

    \item Find the gradient of the curve $y = 3x^2 - 1$ at the point $A(-1, 2)$. \sol{}
          \begin{flalign*}
              \Delta y                                        & = 3{(x + \Delta x)}^2 - 1 - (3x^2 - 1)             \\
                                                              & = 3x^2 + 6x\Delta x + 3{\Delta x}^2 - 1 - 3x^2 + 1 \\
                                                              & = 6x\Delta x + 3{\Delta x}^2                       \\
              \\
              \frac{\Delta y}{\Delta x}                       & = 6x + 3\Delta x                                   \\
              \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} (6x + 3\Delta x)    \\
              \frac{dy}{dx}                                   & = 6x
          \end{flalign*}
          $\therefore$ The gradient of the curve $y = 3x^2 - 1$ at the point $A(-1, 2)$ is $6(-1) = -6$.

    \item Find the gradient of the curve $y = 2x - x^3$ at the point $B(-1, -1)$. \sol{}
          \begin{flalign*}
              \Delta y                                        & = 2(x + \Delta x) - {(x + \Delta x)}^3 - (2x - x^3)                              \\
                                                              & = 2x + 2\Delta x - x^3 - 3x^2\Delta x - 3x{\Delta x}^2 - {\Delta x}^3 - 2x + x^3 \\
                                                              & = 2\Delta x - 3x^2\Delta x - 3x{\Delta x}^2 - {\Delta x}^3                       \\
              \\
              \frac{\Delta y}{\Delta x}                       & = 2 - 3x^2 - 3x\Delta x - {\Delta x}^2                                           \\
              \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} (2 - 3x^2 - 3x\Delta x - {\Delta x}^2)            \\
              \frac{dy}{dx}                                   & = 2 - 3x^2
          \end{flalign*}
          $\therefore$ The gradient of the curve $y = 2x - x^3$ at the point $B(-1, -1)$ is $2 - 3(-1)^2 = -1$.

    \item Find the derivative of the following functions using the definition of the
          derivative, and find the value of the derivative at the point where $x = 1$:
          \begin{enumerate}
              \item $f (x) = x^2 + 2x$
                    \sol{}
                    \begin{flalign*}
                        \Delta y                                        & = {(x + \Delta x)}^2 + 2(x + \Delta x) - (x^2 + 2x)           \\
                                                                        & = x^2 + 2x\Delta x + {\Delta x}^2 + 2x + 2\Delta x - x^2 - 2x \\
                                                                        & = 2x\Delta x + {\Delta x}^2 + 2\Delta x                       \\
                        \\
                        \frac{\Delta y}{\Delta x}                       & = 2x + {\Delta x} + 2                                         \\
                        \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} (2x + {\Delta x} + 2)          \\
                        f'(x)                                           & = 2x + 2
                    \end{flalign*}

              \item $g(x) = x^3$
                    \sol{}
                    \begin{flalign*}
                        \Delta y                                        & = {(x + \Delta x)}^3 - x^3                                          \\
                                                                        & = x^3 + 3x^2\Delta x + 3x{\Delta x}^2 + {\Delta x}^3 - x^3          \\
                                                                        & = 3x^2\Delta x + 3x{\Delta x}^2 + {\Delta x}^3                      \\
                        \\
                        \frac{\Delta y}{\Delta x}                       & = 3x^2 + 3x{\Delta x} + {\Delta x}^2                                \\
                        \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} (3x^2 + 3x{\Delta x} + {\Delta x}^2) \\
                        g'(x)                                           & = 3x^2
                    \end{flalign*}

              \item $h(x) = \dfrac{5}{x}$
                    \sol{}
                    \begin{flalign*}
                        \Delta y                                        & = \frac{5}{x + \Delta x} - \frac{5}{x}                    \\
                                                                        & = \frac{5x - 5(x + \Delta x)}{x(x + \Delta x)}            \\
                                                                        & = \frac{5x - 5x - 5\Delta x}{x(x + \Delta x)}             \\
                                                                        & = \frac{-5\Delta x}{x(x + \Delta x)}                      \\
                        \\
                        \frac{\Delta y}{\Delta x}                       & = \frac{-5}{x(x + \Delta x)}                              \\
                        \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} \frac{-5}{x(x + \Delta x)} \\
                        h'(x)                                           & = -\frac{5}{x^2}
                    \end{flalign*}

              \item $k(x) = \sqrt{x + 3}$
                    \sol{}
                    \begin{flalign*}
                        \Delta y                                         & = \sqrt{x + 3 + \Delta x} - \sqrt{x + 3}                                                                                                                \\
                        \\
                        \dfrac{\Delta y}{\Delta x}                       & = \dfrac{\sqrt{x + 3 + \Delta x} - \sqrt{x + 3}}{\Delta x}                                                                                              \\
                                                                         & = \dfrac{\sqrt{x + 3 + \Delta x} - \sqrt{x + 3}}{\Delta x} \cdot \dfrac{\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}}{\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}} \\
                                                                         & = \dfrac{x + 3 + \Delta x - (x + 3)}{\Delta x\left(\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}\right)}                                                       \\
                                                                         & = \dfrac{\Delta x}{\Delta x\left(\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}\right)}                                                                         \\
                                                                         & = \dfrac{1}{\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}}                                                                                                     \\
                        \\
                        \lim_{\Delta x \to 0} \dfrac{\Delta y}{\Delta x} & = \lim\limits_{\Delta x \to 0} \dfrac{1}{\sqrt{x + 3 + \Delta x} + \sqrt{x + 3}}                                                                        \\
                        k'(x)                                            & = \dfrac{1}{2\sqrt{x + 3}}
                    \end{flalign*}
          \end{enumerate}

                           & = \frac{1}{3}x^{-\frac{2}{3}} + \frac{\sqrt{3}}{6}x^{-\frac{3}{2}}          \\

              \item $y=\left(x^{3}-4x\right)\left(x^{2}+3x-1\right)$
                    \sol{}

                           & = 3x^4 + 9x^3 - 3x^2 - 4x^2 - 12x + 4 + 2x^4 + 3x^3 - 8x^2 - 12x \\
                           & = 5x^4 + 12x^3 - 15x^2 - 24x + 4
                    \end{flalign*}

              \item $y={(x-1)}^{5}{\sqrt{x+2}}$

                           & = (x - 1)^4\left[5\sqrt{x + 2} + \frac{(x-1)}{2\sqrt{x + 2}}\right] \\
                           & = (x - 1)^4\left[\dfrac{10(x + 2) + x - 1}{2\sqrt{x + 2}}\right]    \\
                           & = \dfrac{(x - 1)^4(11x + 19)}{2\sqrt{x + 2}}
                    \end{flalign*}
              \item $y=\left(2x+5\right)\left(x^{2}-2\right)\left(x^{3}-1\right)$
                    \sol{}
                    \begin{flalign*}
                        y' & = \left[(2x^3 - 4x + 5x^2 - 10)(x^3 - 1)\right]'                        \\
                           & = (2x^3 - 4x + 5x^2 - 10)'(x^3 - 1) + (2x^3 - 4x + 5x^2 - 10)(x^3 - 1)' \\
                           & = (6x^2 - 4 + 10x)(x^3 - 1) + (2x^3 - 4x + 5x^2 - 10)(3x^2)             \\
                           & = 6x^5 - 4x^3 + 10x^4 - 6x^2 + 4 - 10x + 6x^5 - 12x^3 + 15x^4 - 30x^2   \\
                           & = 12x^5 + 25x^4 - 16x^3 - 36x^2 - 10x + 4
                    \end{flalign*}

              \item $y={\dfrac{2x^{3}-3x^{2}+4}{x^{2}}}$
                    \sol{}
                    \begin{flalign*}
                        y' & = \frac{(2x^3 - 3x^2 + 4)'(x^2) - (2x^3 - 3x^2 + 4)(x^2)'}{x^4} \\
                           & = \frac{(6x^2 - 6x)(x^2) - (2x^3 - 3x^2 + 4)(2x)}{x^4}          \\
                           & = \frac{6x^4 - 6x^3 - 4x^4 + 6x^3 - 8x}{x^4}                    \\
                           & = \frac{2x^4 - 8x}{x^4}                                         \\
                           & = \frac{2x^3 - 8}{x^3}                                          \\
                           & = 2 - \frac{8}{x^3}
                    \end{flalign*}

              \item $y={\dfrac{x^{2}+4}{x+1}}$
                    \sol{}
                    \begin{flalign*}
                        y' & = \frac{(x^2 + 4)'(x + 1) - (x^2 + 4)(x + 1)'}{(x + 1)^2} \\
                           & = \frac{(2x)(x + 1) - (x^2 + 4)}{(x + 1)^2}               \\
                           & = \frac{2x^2 + 2x - x^2 - 4}{(x + 1)^2}                   \\
                           & = \frac{x^2 + 2x - 4}{(x + 1)^2}
                    \end{flalign*}

              \item $y={\dfrac{x+2}{x^{2}+5x+6}}$
                    \sol{}
                    \begin{flalign*}
                        y' & = \frac{(x + 2)'(x^2 + 5x + 6) - (x + 2)(x^2 + 5x + 6)'}{(x^2 + 5x + 6)^2} \\
                           & = \frac{(x^2 + 5x + 6) - (x + 2)(2x + 5)}{(x^2 + 5x + 6)^2}                \\
                           & = \frac{x^2 + 5x + 6 - 2x^2 - 9x - 10}{(x^2 + 5x + 6)^2}                   \\
                           & = \frac{-x^2 - 4x - 4}{(x^2 + 5x + 6)^2}                                   \\
                           & = -\dfrac{(x+2)^2}{(x+2)^2(x+3)^2}                                         \\
                           & = -\dfrac{1}{(x+3)^2}
                    \end{flalign*}

              \item $y={\dfrac{x^{2}}{{\left(x^{2}-1\right)}^{3}}}$
                    \sol{}
                    \begin{flalign*}
                        y' & = \frac{(x^2)'(x^2 - 1)^3 - x^2\left[(x^2 - 1)^3\right]'}{(x^2 - 1)^6}   \\
                           & = \frac{(2x)(x^2 - 1)^3 - x^2\left[3(x^2 - 1)^2(2x)\right]}{(x^2 - 1)^6} \\
                           & = \frac{2x(x^2 - 1)^3 - 6x^3(x^2 - 1)^2}{(x^2 - 1)^6}                    \\
                           & = \frac{2x(x^2 - 1)^2(x^2 - 1 - 3x^2)}{(x^2 - 1)^6}                      \\
                           & = \frac{2x(x^2 - 1)^2(-2x^2 - 1)}{(x^2 - 1)^6}                           \\
                           & = \frac{-2x(2x^2 + 1)}{(x^2 - 1)^4}                                      \\
                           & = \frac{-4x^3 - 2x}{(x^2 - 1)^4}
                    \end{flalign*}
          \end{enumerate}

    \item Find the derivative of the following functions:

    \item If the function $y = \dfrac{x^3}{{(x-1)}^2}$, find $y'$ and $y''$. \sol{}

    \item Given the function $y = {(2-3x^2)}^4$, find the value of $\dfrac{d^2y}{dx^2}$

    \item Find the derivative $\dfrac{dy}{dx}$ of the following implicit functions:

              \item $x^{2}+2y=2x+3$
                    \sol{}
                    \begin{flalign*}
                        x^{2}+2y & = 2x+3            \\
                        2x + 2y' & = 2               \\
                        y'       & = \dfrac{2-2x}{2} \\
                                 & = 1-x
                    \end{flalign*}

              \item $x^{2}+3x=y^{2}-5y$
                    \sol{}
                    \begin{flalign*}
                        x^{2}+3x   & = y^{2}-5y           \\
                        2x + 3     & = 2y y' - 5y'        \\
                        (2y - 5)y' & = 2x + 3             \\
                        y'         & = \dfrac{2x+3}{2y-5}
                    \end{flalign*}

              \item $3x^{2}+7x y-9y^{2}=2$
                    \sol{}
                    \begin{flalign*}
                        3x^{2}+7x y-9y^{2}     & = 2                          \\
                        6x + 7y + 7xy' - 18yy' & = 0                          \\
                        7xy' - 18yy'           & = -6x - 7y                   \\
                        y'(7x - 18y)           & = -6x - 7y                   \\
                        y'                     & = -\dfrac{6x + 7y}{7x - 18y}
                    \end{flalign*}

              \item $x^{3}y+x y^{3}=3x y$
                    \sol{}

                        3x^2y + x^3y' + y^3 + 3xy^2y' & = 3y + 3xy'                                    \\
                        x^3y' + 3xy^2y' - 3xy'        & = 3y - 3x^2y - y^3                             \\

                        y'                            & = \dfrac{3y - 3x^2y - y^3}{x^3 + 3xy^2 - 3x}   \\
                                                      & = \dfrac{y(3 - 3x^2 - y^2)}{x(x^2 + 3y^2 - 3)}
                    \end{flalign*}
          \end{enumerate}

    \item Find the gradient of the tangent to the curve $x^2 + xy + y^2 = 4$ at the point
          $A(2, -2)$. \sol{}
          \begin{flalign*}
              x^2 + xy + y^2      & = 4                       \\
              2x + y + xy' + 2yy' & = 0                       \\
              xy' + 2yy'          & = -2x - y                 \\
              (x + 2y)y'          & = -2x - y                 \\
              y'                  & = -\dfrac{2x + y}{x + 2y} \\
          \end{flalign*}

          \begin{flalign*}
              y' & = -\dfrac{2(2) + (-2)}{2 + 2(-2)} \\
                 & = -\dfrac{2}{-2}                  \\
                 & = 1
          \end{flalign*}
