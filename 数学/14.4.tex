        \item $(x-3)(4-x)$ and $(6-x)(x-1)$
              \sol{}
              \begin{flalign*}
                             & (x-3)(4-x) - (6-x)(x-1)            \\
                             & = -x^2 + 7x - 12 - (-x^2 + 7x - 6) \\
                             & = -x^2 + 7x - 12 + x^2 - 7x + 6    \\
                             & = -6 < 0                           \\
                  \therefore & \  (x-3)(4-x) < (6-x)(x-1)
              \end{flalign*}
        \item $6 - x^2$ and $4x - 2x^2$
              \sol{}
              \begin{flalign*}
                             & 6 - x^2 - (4x - 2x^2)  \\
                             & = 6x - x^2 - 4x + 2x^2 \\
                             & = x^2 - 2x             \\
                             & = {(x - 1)}^2 + 1      \\
                  \because   & \ {(x - 1)}^2 + 1 > 0  \\
                  \therefore & \ {(x - 1)}^2 + 1 > 0  \\
                  \therefore & \ 6 - x^2 > 4x - 2x^2
              \end{flalign*}
    \end{enumerate}
