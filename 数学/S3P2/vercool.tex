\section*{MIT Integration Bee 2023 Regular Season}
\subsection*{Problem 8}
\begin{flalign*}
     & \int_{0}^{\pi}x\sin^4(x)dx                                                                                                                  & \\
     & \text{Let } x = \pi - u,\ u = \pi - x,\ dx = -du                                                                                              \\
     & \text{When } x = 0,\ u = \pi,\ \text{and when } x = \pi,\ u = 0                                                                               \\
     & = \int_{\pi}^{0}(\pi - u)\sin^4(\pi - u)(-du)                                                                                                 \\
     & = \int_{0}^{\pi}(\pi - u)\sin^4(\pi - u)du                                                                                                    \\
     & = \int_{0}^{\pi}(\pi - u)\sin^4(u)du                                                                                                          \\
     & = \int_{0}^{\pi}\pi\sin^4(u)du - \int_{0}^{\pi}u\sin^4(u)du                                                                                   \\
     & = \int_{0}^{\pi}\pi\sin^4(u)du - I                                                                                                            \\
     & = \dfrac{\pi}{2}\int_{0}^{\pi}\sin^4(u)du                                                                                                     \\
     & = \dfrac{\pi}{2}\int_{0}^{\pi}{\left(\dfrac{1 - \cos2u}{2}\right)}^2du                                                                        \\
     & = \dfrac{\pi}{8}\int_{0}^{\pi}{\left(1 - 2\cos2u + \cos^2 2u\right)}du                                                                        \\
     & = \dfrac{\pi}{8}\left(\int_{0}^{\pi}du - 2\int_{0}^{\pi}\cos(2u)du + \int_{0}^{\pi}\cos^2 (2u)du\right)                                       \\
     & = \dfrac{\pi}{8}\left(\pi - 2\int_{0}^{\pi}\cos(2u)du + \int_{0}^{\pi}\dfrac{1 + \cos(4u)}{2}du\right)                                        \\
     & = \dfrac{\pi}{8}\left(\pi - 2\cdot {\left[\dfrac{1}{2}\sin(2u)\right]}_0^\pi + \dfrac{1}{2}\int_{0}^{\pi}\left[1 + \cos(4u)\right]du\right)   \\
     & = \dfrac{\pi}{8}\left(\pi - 2\cdot (0 - 0) + \dfrac{1}{2}\int_{0}^{\pi}du + \dfrac{1}{2}\int_{0}^{\pi}\cos(4u)du\right)                       \\
     & = \dfrac{\pi}{8}\left(\pi + \dfrac{1}{2}\pi + \dfrac{1}{2}\cdot {\left[\dfrac{1}{4}\sin(4u)\right]}_0^\pi\right)                              \\
     & = \dfrac{\pi}{8}\left(\dfrac{3}{2}\pi + \dfrac{1}{2}\cdot(0 - 0)\right)                                                                       \\
     & = \dfrac{3}{16}\pi
\end{flalign*}
