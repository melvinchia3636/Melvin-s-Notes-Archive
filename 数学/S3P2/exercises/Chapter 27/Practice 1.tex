\subsection{Practice 1}
\noindent \hspace{1.2em}\textit{Find the following indefinite integrals:}
\begin{enumerate}
    \begin{multicols}{2}
        \item $\displaystyle\int x^{3}dx$
        \sol{}
        \begin{flalign*}
            I & = \dfrac{1}{4}x^{4} + C &
        \end{flalign*}
        \vfill{}\null{}

        \item $\displaystyle\int{\sqrt[3]{x}}dx$
        \sol{}
        \begin{flalign*}
            I & = \int{x^{\frac{1}{3}}}dx         & \\
              & = \dfrac{3}{4}x^{\frac{4}{3}} + C
        \end{flalign*}
    \end{multicols}
    \begin{multicols}{2}
        \item $\displaystyle\int{\dfrac{1}{x^{3}}}dx$
        \sol{}
        \begin{flalign*}
            I & = \int{x^{-3}}dx          & \\
              & = -\dfrac{1}{2}x^{-2} + C
        \end{flalign*}

        \item $\displaystyle\int{\dfrac{1}{\sqrt[3]{x}}}dx$
        \sol{}
        \begin{flalign*}
            I & = \int{x^{-\frac{1}{3}}}dx & \\
              & = 3x^{\frac{2}{3}} + C
        \end{flalign*}
    \end{multicols}
\end{enumerate}
