\subsection{Practice 3}

\noindent \hspace{1.2em}\textit{
    Find the extreme values of the following functions (Question 1 to 4):
}

\begin{enumerate}
    \begin{multicols}{2}
        \item $f(x)=x^2+x-6$
        \sol{}
        \begin{flalign*}
            f'(x)                         & = 2x + 1        \\
            2x + 1                        & = 0             \\
            x                             & = -\dfrac{1}{2} \\
            f''(x)                        & = 2             \\
            f''\left(-\dfrac{1}{2}\right) & = 2 > 0
        \end{flalign*}
        When $x = -\dfrac{1}{2}$, $f(x)$ has a minimum value of $-\dfrac{25}{4}$.

        \item $f(x)=2-x-x^2$
        \sol{}
        \begin{flalign*}
            f'(x)                         & = -2x - 1       \\
            -2x - 1                       & = 0             \\
            x                             & = -\dfrac{1}{2} \\
            f''(x)                        & = -2            \\
            f''\left(-\dfrac{1}{2}\right) & = -2 < 0
        \end{flalign*}
        When $x = -\dfrac{1}{2}$, $f(x)$ has a maximum value of $\dfrac{9}{4}$.
    \end{multicols}
    \newpage

    \begin{multicols}{2}
        \item $f(x)=\dfrac{1}{3} x^3-\dfrac{1}{2} x^2-2x+2$
        \sol{}
        \begin{flalign*}
            f'(x)   & = x^2 - x - 2          \\
                    & = (x - 2)(x + 1)       \\
            f''(x)  & = 2x - 1               \\
                    & = 2(x - \dfrac{1}{2})  \\
            f''(-1) & = 2(-1 - \dfrac{1}{2}) \\
                    & = -\dfrac{3}{2} < 0    \\
            f''(2)  & = 2(2 - \dfrac{1}{2})  \\
                    & = \dfrac{7}{2} > 0
        \end{flalign*}
        $f(x)$ has a maximum value of $\dfrac{19}{6}$.

        $f(x)$ has a minimum value of $-\dfrac{4}{3}$.
        \vfill\null
        \columnbreak
        \item $f(x)=4 x-3 x^3$
        \sol{}
        \begin{flalign*}
            f'(x)                         & = 4 - 9x^2         \\
            4 - 9x^2                      & = 0                \\
            x                             & = \pm \dfrac{2}{3} \\
            f''(x)                        & = -18x             \\
            f''\left(\dfrac{2}{3}\right)  & = -12 < 0          \\
            f''\left(-\dfrac{2}{3}\right) & = 12 > 0
        \end{flalign*}
        When $x = \dfrac{2}{3}$, $f(x)$ has a maximum value of $\dfrac{16}{9}$.

        When $x = -\dfrac{2}{3}$, $f(x)$ has a minimum value of $-\dfrac{16}{9}$.
    \end{multicols}
\end{enumerate}

\noindent \hspace{1.2em}\textit{Find the coordinates of the extreme points of the following functions (Question 5 to 6):}
\begin{enumerate}[resume]
    \begin{multicols}{2}
        \item $y=2 x^3-3 x^2-12 x-7$
        \sol{}
        \begin{flalign*}
            y'             & = 6x^2 - 6x - 12 \\
            x^2 - x - 2    & = 0              \\
            (x - 2)(x + 1) & = 0              \\
            x = 2          & \text{ or } -1   \\
            y''            & = 12x - 6        \\
            y''(2)         & = 18 > 0         \\
            y''(-1)        & = -18 < 0
        \end{flalign*}
        $(2, - 27)$ is a minimum point.

        $(-1, 0)$ is a maximum point.

        \item $y=x+\dfrac{1}{x}$
        \sol{}
        \begin{flalign*}
            y'                 & = 1 - \dfrac{1}{x^2} \\
            1 - \dfrac{1}{x^2} & = 0                  \\
            x                  & = \pm 1              \\
            y''                & = \dfrac{2}{x^3}     \\
            y''(1)             & = 2 > 0              \\
            y''(-1)            & = -2 < 0
        \end{flalign*}
        $(1, 2)$ is a minimum point.

        $(-1, -2)$ is a maximum point.
    \end{multicols}
\end{enumerate}
