    \section{Kertas 1}

    \begin{enumerate}
        \item Given that $\displaystyle \int_m^2(2x+3)\,dx = -8$ where $m > 0$, find the
              value of $m$. \sol{}
              \begin{flalign*}
                  \int_m^2(2x+3)\,dx & = \left[x^2 + 3x\right]_m^2 \\
                                     & = 4 + 6 - m^2 - 3m          \\
                                     & = 10 - 3m - m^2             \\
                  10 - 3m - m^2      & = -8                        \\
                  m^2 + 3m - 10      & = 8                         \\
                  m^2 + 3m - 18      & = 0                         \\
                  (m+6)(m-3)         & = 0                         \\
                  m                  & = 3 \quad (m > 0) \eos
              \end{flalign*}

        \item Given that $\dfrac{dy}{dx} = 10{(5x + 3)}^2$ and $y = 4$ when $x = 0$. Express
