    \item Find the height from vertex B to side AC in $\Delta ABC$, where the vertices
          are $A(3, 6), B(2, -3)$, and $C(-1, 4)$. \newline\newline \noindent
          \textbf{Sol{}.}
          \begin{flalign*}
              m_{AC}     & = \frac{y_2 - y_1}{x_2 - x_1}                           \\
                         & = \frac{4 - 6}{-1 - 3}                                  \\
                         & = \frac{-2}{-4}                                         \\
                         & = \frac{1}{2}                                           \\
              y - y_A    & = m_{AC}(x - x_A)                                       \\
              y - 6      & = \frac{1}{2}(x - 3)                                    \\
              2y - 12    & = x-3                                                   \\
              x - 2y + 9 & = 0                                                     \\
              d          & = \left|\frac{Ax_B + By_B + C}{\sqrt{A^2 + B^2}}\right| \\
                         & = \left|\frac{2 + 6 + 9}{\sqrt{1^2 + {(-2)}^2}}\right|  \\
                         & = \left|\frac{17}{\sqrt{5}}\right|                      \\
                         & = \frac{17\sqrt5}{5}                                    \\
          \end{flalign*}

    \item Find the vertices of triangle whose mid point of sides are (-2, -5), (-1, 1)
          and (4, -1). \newline\newline \noindent \textbf{Sol{}.}

          Let the vertices be $A(x_1, y_1)$, $B(x_2, y_2)$ and $C(x_3, y_3)$

          \begin{flalign*}
              \frac{x_1 + x_2}{2}                                                  & = -2  & (1)  \\
              \frac{x_2 + x_3}{2}                                                  & = -1  & (2)  \\
              \frac{x_3 + x_1}{2}                                                  & = 4   & (3)  \\
              \\
              (1) \Rightarrow x_1 + x_2                                            & = -4  & (4)  \\
              (2) \Rightarrow x_2 + x_3                                            & = -2  & (5)  \\
              (3) \Rightarrow x_3 + x_1                                            & = 8   & (6)  \\
              \\
              (4) + (5) + (6) \Rightarrow 2x_1 + 2x_2 + 2x_3                       & = 2          \\
              x_1 + x_2 + x_3                                                      & = 1   & (7)  \\
              \\
              \text{Sub (4) in (7)}                           \Rightarrow -4 + x_3 & = 1          \\
              x_3                                                                  & = 5          \\
              \text{Sub $x_3 = 5$ in (3)} \Rightarrow x_1                          & = 3          \\
              \text{Sub $x_3 = 5$ in (2)} \Rightarrow x_2                          & = -7         \\
              \\
              \frac{y_1 + y_2}{2}                                                  & = -5  & (8)  \\
              \frac{y_2 + y_3}{2}                                                  & = 1   & (9)  \\
              \frac{y_3 + y_1}{2}                                                  & = -1  & (10) \\
              \\
              (8) \Rightarrow y_1 + y_2                                            & = -10 & (11) \\
              (9) \Rightarrow y_2 + y_3                                            & = 2   & (12) \\
              (10) \Rightarrow y_3 + y_1                                           & = -2  & (13) \\
              \\
              (11) + (12) + (13) \Rightarrow 2y_1 + 2y_2 + 2y_3                    & = -10        \\
              y_1 + y_2 + y_3                                                      & = -5  & (14) \\
              \\
              \text{Sub (11) in (14)} \Rightarrow -10 + y_3                        & = -5  & (15) \\
              y_3                                                                  & = 5   & (16) \\
              \text{Sub $y_3 = 5$ in (13)} \Rightarrow y_1                         & = -7  & (17) \\
              \text{Sub $y_3 = 5$ in (12)} \Rightarrow y_2                         & = -3  & (18) \\
              \\
              \therefore\ A(-7, -3),\ B(3, -7) \text{ and } C(5, 5)
          \end{flalign*}

    \item If (-1, -2), (2, t), and (3, 6) are on the same line, find the value of t.
          \newline\newline \noindent \textbf{Sol{}.}
          \begin{flalign*}
              \frac{1}{2}\left|(-t + 12 -6)-(-4+3t-6)\right| & = 0   \\
              -t + 6 + 10 - 3t                               & =     \\
              -4t                                            & = -16 \\
              t                                              & = 4
          \end{flalign*}

\end{enumerate}
