\section{Arithmetic Properties of Limits}

\begin{mdframed}[style=MyFrame]
      If $\lim\limits_{x \to x_0} f(x) = A$, $\lim\limits_{x \to x_0} g(x) = B$, then:
      \begin{enumerate}[label= (\alph*)]
            \item $\lim\limits_{x \to x_0} (f(x) \pm g(x)) = \lim\limits_{x \to x_0} f(x) \pm \lim\limits_{x \to x_0} g(x) = A \pm B$
            \item $\lim\limits_{x \to x_0} (f(x) \cdot g(x)) = \lim\limits_{x \to x_0} f(x) \cdot \lim\limits_{x \to x_0} g(x) = A \cdot B$
            \item $\lim\limits_{x \to x_0} \left[\dfrac{f(x)}{g(x)}\right] = \dfrac{\lim\limits_{x \to x_0} f(x)}{\lim\limits_{x \to x_0} g(x)} = \dfrac{A}{B}$, ($B \neq 0$)
            \item If $k$ is a constant, then $\lim\limits_{x \to x_0} k = k$
            \item If $k$ is a constant, then $\lim\limits_{x \to x_0} k \cdot f(x) = k
                        \lim\limits_{x \to x_0} f(x) = kA$
            \item If $n \in \mathbb{R}$, and $\lim\limits_{x \to x_0} f(x) > 0$, then
                  $\lim\limits_{x \to x_0} {[f(x)]}^n = {\left[\lim\limits_{x \to
                                    x_0}f(x)\right]}^n = A^n$
            \item If $\lim\limits_{x \to x_0} f(x) = 0$, then $\lim\limits_{x \to x_0}
                        \dfrac{1}{f(x)} = \infty$
            \item L'Hopital's Rule: If $\lim\limits_{x \to x_0} \dfrac{f(x)}{g(x)}$ is
                  indeterminate, then $\lim\limits_{x \to x_0} \dfrac{f(x)}{g(x)} =
                        \lim\limits_{x \to x_0} \dfrac{f'(x)}{g'(x)}$
      \end{enumerate}
\end{mdframed}

\subsection*{Squeeze Theorem or Sandwich Rule}

\begin{mdframed}[style=MyFrame]
      \begin{flalign*}
                         & \text{Near point } x_0,                                          & \\
            \text{If }   & f(x) \leq g(x) \leq h(x)                                           \\
            \text{and }  & \lim\limits_{x \to x_0} f(x) = \lim\limits_{x \to x_0} h(x) = A, & \\
            \text{then } & \lim\limits_{x \to x_0} g(x) = A.                                & \\
      \end{flalign*}
\end{mdframed}

      \begin{flalign*}
            \text{Hence} \qquad & \dfrac{1}{2} \times 1 \times \sin{x} < \dfrac{1}{2} \times 1^2 < \dfrac{1}{2} \times 1 \times \tan{x} & \\
            \therefore \qquad   & \sin{x} < x < \tan{x}
      \end{flalign*}
      Dividing by $\sin{x}\ (\sin{x} \neq 0$), we get
      \begin{flalign*}
            \quad \quad            & 1 < \dfrac{x}{\sin{x}} < \dfrac{1}{\cos{x}}                               & \\
            \text{But} \quad \quad & \lim_{x\to0^+}{\dfrac{1}{\cos{x}}} = \dfrac{x}{\lim_{x\to0^+}\sin{x}} = 1
      \end{flalign*}
      Hence, $\lim\limits_{x\to0^+}{\dfrac{x}{\sin{x}}}$ is in between $1$ and $\lim\limits_{x\to0^+}{\dfrac{1}{\cos{x}}}$, which is in between $1$ and $1$. Therefore, $\lim\limits_{x\to0^+}{\dfrac{x}{\sin{x}}}$ must be equal to $1$ (Squeeze Theorem).
      \begin{flalign*}
            \text{That is} \qquad & \lim\limits_{x\to0^+}{\dfrac{x}{\sin{x}}} = 1,                                                                                                                     & \\
            \therefore \qquad     & \lim\limits_{x\to0^+}{\dfrac{\sin{x}}{x}} = \lim_{x\to0^+}{\dfrac{1}{\dfrac{x}{\sin{x}}}} = \lim_{x\to0^+}{\dfrac{1}{\lim_{x\to0^+}{\dfrac{x}{\sin{x}}}}} = 1 \eos
      \end{flalign*}
